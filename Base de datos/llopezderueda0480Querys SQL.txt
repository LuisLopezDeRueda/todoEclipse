Primera 

SELECT film.title, actor.first_name, actor.last_name FROM actor JOIN film_actor ON actor.actor_id = film_actor.actor_id JOIN film ON film_actor.film_id = film.film_id WHERE actor.first_name LIKE 'MICHAEL' AND film.`length` > 120;

Segunda

SELECT customer.first_name, customer.last_name , SUM(payment.amount) FROM payment left JOIN customer ON payment.customer_id = customer.customer_id GROUP BY payment.customer_id;

Tercera

SELECT country.country,COUNT(city.city) FROM country LEFT JOIN city ON country.country_id= city.country_id GROUP BY country ORDER BY COUNT(city.city) ASC;
